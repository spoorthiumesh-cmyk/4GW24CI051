#include <stdio.h>

/* Function to find waiting time */
void findWaitingTime(int bt[], int n, int wt[]) {
    wt[0] = 0;
    for (int i = 1; i < n; i++) {
        wt[i] = bt[i - 1] + wt[i - 1];
    }
}

/* Function to find turnaround time */
void findTurnAroundTime(int bt[], int n, int wt[], int tat[]) {
    for (int i = 0; i < n; i++) {
        tat[i] = bt[i] + wt[i];
    }
}

/* Function to calculate average times */
void findAvgTime(int p[], int bt[], int n) {
    int wt[n], tat[n];
    int total_wt = 0, total_tat = 0;

    findWaitingTime(bt, n, wt);
    findTurnAroundTime(bt, n, wt, tat);

    printf("\nProcess\tBT\tWT\tTAT\n");
    for (int i = 0; i < n; i++) {
        total_wt += wt[i];
        total_tat += tat[i];
        printf("P%d\t%d\t%d\t%d\n", p[i], bt[i], wt[i], tat[i]);
    }

    printf("\nAverage Waiting Time = %.2f", (float)total_wt / n);
    printf("\nAverage Turnaround Time = %.2f\n", (float)total_tat / n);
}

/* Sort processes by priority (lower number = higher priority) */
void sortByPriority(int p[], int bt[], int pr[], int n) {
    for (int i = 0; i < n; i++) {
        for (int j = i + 1; j < n; j++) {
            if (pr[i] > pr[j]) {   // Correct comparison
                int temp;

                temp = pr[i];
                pr[i] = pr[j];
                pr[j] = temp;

                temp = p[i];
                p[i] = p[j];
                p[j] = temp;

                temp = bt[i];
                bt[i] = bt[j];
                bt[j] = temp;
            }
        }
    }
}

/* Priority Scheduling */
void priorityScheduling(int p[], int bt[], int pr[], int n) {
    sortByPriority(p, bt, pr, n);

    printf("\nExecution Order: ");
    for (int i = 0; i < n; i++) {
        printf("P%d ", p[i]);
    }
    printf("\n");

    findAvgTime(p, bt, n);
}

/* Driver Code */
int main() {
    int n;
    printf("Enter number of processes: ");
    scanf("%d", &n);

    int p[n], bt[n], pr[n];

    for (int i = 0; i < n; i++) {
        p[i] = i + 1;
        printf("\nEnter Burst Time for P%d: ", p[i]);
        scanf("%d", &bt[i]);
        printf("Enter Priority for P%d: ", p[i]);
        scanf("%d", &pr[i]);
    }

    priorityScheduling(p, bt, pr, n);

    return 0;
}
